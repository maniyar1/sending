fn send_to_all(teams: Vec<&HashMap<String, Player>>, text: String) {
    for map in teams {
        for (_name, player) in map {
            let mut stream = player.stream.lock().unwrap();
            stream.write(text.as_bytes()).unwrap();
        }
    }
